ek2_set_height_race_flags_effect = {
	# Reset local EK2 height race flags so recalculation stays deterministic.
	remove_character_flag = race_ek2_height_goblinken
	remove_character_flag = race_ek2_height_altmer_ayleid
	remove_character_flag = race_ek2_height_nord
	remove_character_flag = race_ek2_height_giant_minotaur
	remove_character_flag = race_ek2_height_very_tall
	remove_character_flag = race_ek2_height_tall
	remove_character_flag = race_ek2_height_short
	remove_character_flag = race_ek2_height_average

	# Goblinken: shortest.
	if = {
		limit = {
			OR = {
				has_character_flag = race_goblinken
				race = race_goblin_green
				race = race_goblin_yellow
				race = race_goblin_riekling
				race = race_goblin_riekr
			}
		}
		add_character_flag = race_ek2_height_goblinken
	}

	# Altmer/Ayleid: tallest mer.
	else_if = {
		limit = {
			OR = {
				race = race_altmer
				race = race_altmer_pure_1
				race = race_altmer_pure_2
				race = race_altmer_ear
				race = race_altmer_balfieri_1
				race = race_altmer_balfieri_2
				race = race_ayleid_1
				race = race_ayleid_2
				race = race_ayleid_3
			}
		}
		add_character_flag = race_ek2_height_altmer_ayleid
	}

	# Giants/Minotaurs: tallest possible.
	else_if = {
		limit = {
			OR = {
				has_character_flag = race_giant
				has_character_flag = race_minotaur
				race = race_giant
				race = race_giant_kamal
				race = race_minotaur
				race = race_sea_giant
			}
		}
		add_character_flag = race_ek2_height_giant_minotaur
	}

	# Very tall races.
	else_if = {
		limit = {
			OR = {
				has_character_flag = race_daedra
				has_character_flag = race_imga
				has_character_flag = race_kapotun
				race = race_orsimer
				race = race_orsimer_2
				race = race_orsimer_iron
				race = race_orsimer_malahk
				race = race_maormer
				race = race_dremora
				race = race_dremora_eso
				race = race_imga_orangutang
				race = race_imga_gorilla
			}
		}
		add_character_flag = race_ek2_height_very_tall
	}

	# Nords/Atmorans: tall human baseline.
	else_if = {
		limit = {
			OR = {
				race = race_nord_1
				race = race_nord_2
				race = race_nord_3
				race = race_nord_4
				race = race_nord_cyronord_1
				race = race_nord_cyronord_2
				race = race_nord_cyronord_3
				race = race_nord_cyronord_4
				race = race_nord_cyronord_5
			}
		}
		add_character_flag = race_ek2_height_nord
	}

	# Slightly tall races.
	else_if = {
		limit = {
			OR = {
				has_character_flag = race_argonian
				has_character_flag = race_echmer
				race = race_dunmer
				race = race_dunmer_ear
				race = race_dunmer_2
				race = race_dunmer_ashlander
				race = race_argonian
				race = race_argonian_sarpa
				race = race_argonian_naga
				race = race_argonian_paatru
				race = race_tsaesci_1
				race = race_tsaesci_2
				race = race_tsaesci_3
				race = race_tsaesci_4
				race = race_falmer
				race = race_falmer_ear
				race = race_falmer_betrayed
			}
		}
		add_character_flag = race_ek2_height_tall
	}

	# Short races.
	else_if = {
		limit = {
			OR = {
				has_character_flag = race_lilmothiit
				race = race_bosmer
				race = race_lilmothiit
			}
		}
		add_character_flag = race_ek2_height_short
	}

	# Average-height groups.
	else_if = {
		limit = {
			OR = {
				has_character_flag = race_khajiit
				has_character_flag = race_tangmo
				has_character_flag = race_faun
				race = race_khajiit
				race = race_redguard_forebear_1
				race = race_redguard_forebear_2
				race = race_redguard_forebear_3
				race = race_redguard_forebear_4
				race = race_breton_glenumbran_1
				race = race_breton_glenumbran_2
				race = race_breton_glenumbran_3
				race = race_breton_bjoule_1
				race = race_breton_bjoule_2
				race = race_breton_bjoule_3
				race = race_breton_northman_1
				race = race_breton_northman_2
				race = race_imperial_colovian_1
				race = race_imperial_colovian_2
				race = race_imperial_colovian_3
				race = race_imperial_colovian_4
				race = race_imperial_colovian_1_dark
				race = race_imperial_colovian_4_dark
				race = race_imperial_colovian_5_dark
				race = race_imperial_nibenese_1
				race = race_imperial_nibenese_2
				race = race_imperial_nibenese_3
				race = race_imperial_nibenese_4
				race = race_imperial_nibenese_5
				race = race_imperial_nibenese_6
				race = race_imperial_nibenese_1_dark
				race = race_imperial_nibenese_2_dark
				race = race_imperial_nibenese_3_dark
				race = race_imperial_nibenese_4_dark
				race = race_faun
			}
		}
		add_character_flag = race_ek2_height_average
	}
}
